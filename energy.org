usage <- fromJSON("data/agg.json") %>%
  tibble() %>%
  mutate(
    date = parse_datetime(date),
    gas = gas_high + gas_low,
    elec = elec_high + elec_low
  ) %>%
  select(-gas_high, -gas_low, -elec_high, -elec_low) %>%
    filter(
      date > parse_datetime("2022-11-01"),
    date < parse_datetime("2023-05-01"))
